# main.py

import random
import time
from algoritmos import (
    busqueda_lineal,
    busqueda_binaria,
    busqueda_interpolacion
)

def medir_tiempo(funcion, lista, objetivo, nombre):
    inicio = time.time()
    resultado = funcion(lista, objetivo)
    fin = time.time()
    print(f"{nombre}: √çndice {resultado}, Tiempo: {round(fin - inicio, 6)} seg")

def main():
    tamanio = 100_000
    lista = sorted(random.sample(range(1, 1_000_000), tamanio))
    objetivo = random.choice(lista)
    
    print(f"\nüîç Buscando el n√∫mero {objetivo} en una lista de {tamanio} elementos...\n")
    
    medir_tiempo(busqueda_lineal, lista, objetivo, "B√∫squeda Lineal")
    medir_tiempo(busqueda_binaria, lista, objetivo, "B√∫squeda Binaria")
    medir_tiempo(busqueda_interpolacion, lista, objetivo, "B√∫squeda por Interpolaci√≥n")

if __name__ == "__main__":
    main()
